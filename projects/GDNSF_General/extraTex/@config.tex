\newcommand{\GDNSFSetupBundledFonts}{%

\newcommand{\GDNSFSetupLocalFontFiles}[1]{%
  \setCJKmainfont[Path={#1},AutoFakeBold=2.5]{SimSun.ttf}%
  \setCJKfamilyfont{zhsong}[Path={#1},AutoFakeBold=2.5]{SimSun.ttf}%
  \setCJKsansfont[Path={#1},AutoFakeBold=2.5]{NotoSansSC-Bold.otf}%
  \setCJKfamilyfont{zhhei}[Path={#1}]{AdobeHeitiStd-Regular.otf}%
  \setCJKfamilyfont{zhkai}[Path={#1},AutoFakeBold=2.5]{Kaiti.ttf}%
  \setCJKfamilyfont{zhfs}[Path={#1},AutoFakeBold=2.5]{AdobeFangsongStd-Regular.otf}%
  \setmainfont[Path={#1},Extension=.ttf,AutoFakeBold=5,AutoFakeSlant=0.2]{TimesNewRoman}%
}

\newcommand{\GDNSFSetupFonts}{%
  \IfFileExists{styles/fonts/SimSun.ttf}{%
    \GDNSFSetupLocalFontFiles{styles/fonts/}%
  }{%
    \IfFileExists{bensz-fonts.sty}{%
      \usepackage{bensz-fonts}%
      \GDNSFSetupBundledFonts%
    }{%
      \setmainfont{Times New Roman}%
      \setCJKmainfont[AutoFakeBold=2.5]{SimSun}%
      \setCJKfamilyfont{zhsong}[AutoFakeBold=2.5]{SimSun}%
      \setCJKsansfont[AutoFakeBold=2.5]{SimHei}%
      \setCJKfamilyfont{zhhei}[AutoFakeBold=2.5]{SimHei}%
      \setCJKfamilyfont{zhkai}[AutoFakeBold=2.5]{KaiTi}%
      \setCJKfamilyfont{zhfs}[AutoFakeBold=2.5]{FangSong}%
    }%
  }%
  \xeCJKsetup{PunctStyle=quanjiao}%
}
